      let data1 = [11,22,33,44,55]
      let data2 = [66,77,88,99]
      
      document.write(`<h2>Data1 ${data1}</h2>`)
      document.write(`<h2>Data2 ${data2}</h2>`)

      let data =[];

      for(let i=0; i<data1.length; i++){
        data[i]=data1[i];
      }

      for(let i=0; i<data2.length; i++){
        data[data.length+i]=data2[i];
      }
